3. Write a program to swap values of two int variables
#include <stdio.h>

int main(){
    printf("Enter the number \n");
   int number1;
   int number2;
   int number3;
   scanf("%d %d",&number1,&number2);
   number3 = number1;
   number1 = number2;
   number2 = number3;

   printf("after swap number1=%d and number2=%d:",number1,number2);
   return 0;

}
